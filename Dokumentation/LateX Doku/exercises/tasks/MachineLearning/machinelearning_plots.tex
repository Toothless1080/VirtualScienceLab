	\begin{itemize}
		\item Lese aus \lstinline$sklearn.datasets$ den \lstinline$breast_cancer$-Datensatz.

	\end{itemize}

	\begin{itemize}
		\item Lese aus \lstinline$sklearn.datasets$ den \lstinline$breast_cancer$-Datensatz. 
		\item Schränke die Daten auf zwei Spalten ein.
		\item Erzeuge ein neues Histrogramm und vergebe einen Titel und Achsenbeschriftungen.
		\item Zeige den Plot an.
	\end{itemize}

	\begin{itemize}
		\item Lese aus \lstinline$sklearn.datasets$ den \lstinline$breast_cancer$-Datensatz. 
		\item Schränke die Daten auf eine kleinere Menge (z.B. 20) ein.
		\item Erzeuge einen neuen Scatter-Plot und vergebe einen Titel und Achsenbeschriftungen.
		\item Ändere die Farbe der Punkte auf rot.
		\item Ändere die Symbole zu Sternen.
		\item Zeige den Plot an.
